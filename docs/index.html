Quick Start

# Using Docker (Recommended)
docker-compose -f docker-compose.prod.yml up -d

# Or install locally
git clone https://github.com/TwoZer00/FireLab.git
cd FireLab
cd backend && npm install
cd ../frontend && npm install
npm run dev
